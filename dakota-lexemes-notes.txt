objective-c boxing:
  NSNumber* number =         @float/int/oct/hex/binary
  NSNumber* number =         @false
  NSNumber* number =         @true
  NSNumber* character =      @'...'
  NSString* string =         @"..."
  NSArray* array =           @[ id1, id2, ... ]
  NSDictionary* dictionary = @{ id1 : id2, id3 : id4, ... }
                             @( expression )

#| ... | returns a symbol-t (not an object-t)
@| ... | not supported in objective c
